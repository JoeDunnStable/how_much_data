369  inline static complex<RealType> continued_fraction_b(ExpIntFraction g,
370  const RealType& factor,
371  boost::uintmax_t& max_terms)
372  {
373 
374  complex<RealType> tiny = 1.e-100;
375 
376  std::pair<complex<RealType>, complex<RealType> > v = g();
377 
378  complex<RealType> f, C, D, delta;
379  complex<RealType> zero{0};
380  complex<RealType> one{1};
381  f = v.second;
382  if(f == zero)
383  f = tiny;
384  C = f;
385  D = 0;
386 
387  boost::uintmax_t counter(max_terms);
388 
389  do{
390  v = g();
391  D = v.second + v.first * D;
392  if(D == zero)
393  D = tiny;
394  C = v.second + v.first / C;
395  if(C == zero)
396  C = tiny;
397  D = one/D;
398  delta = C*D;
399  f = f * delta;
400  }while((abs(delta - one) > factor) && --counter);
401 
402  max_terms = max_terms - counter;
403 
404  return f;
405  }

418  inline static std::complex<RealType> expint_as_series(RealType n, std::complex<RealType> const& z) {
419  int max_iter = 1000;
420  RealType s = 1-n; // the first argument to the incomplete gamma function
421  complex<RealType> ret;
422  if (s>0 || floor(s) != s) {
423  ret = pow(z,-s) * tgamma(s);
424  complex<RealType> fac = 1;
425  for (int i=0; i<max_iter; ++i){
426  complex<RealType> term = fac / (s+i);
427  ret -= term;
428  if (abs(term) <= abs(ret) * std::numeric_limits<RealType>::epsilon()) break;
429  fac *= -z/static_cast<std::complex<RealType> >(i+1);
430  }
431  } else {
432  complex<RealType> gamma0 = -euler<RealType>()-log(z);
433  complex<RealType> fac = 1;
434  for (int i = 1; i<max_iter; ++i) {
435  fac *= (-z)/static_cast<RealType>(i);
436  complex<RealType> term = fac/static_cast<RealType>(i);
437  gamma0 -= term;
438  if (abs(term) <= abs(gamma0) * std::numeric_limits<RealType>::epsilon())
439  break;
440  }
441  ret = pow(-1,-s) * gamma0 * pow(z,-s);
442  fac = tgamma(-s);
443  complex<double> sum = 0;
444  for (int i=0; i< static_cast<int>(-s); ++i) {
445  complex<RealType> term = fac;
446  sum += term;
447  if (i != -s-1) fac *= (-z)/(-s-i-1);
448  }
449  ret += exp(-z) * sum;
450  ret /= tgamma(-s+1);
451  }
452  if (isnan(ret.real()) || isnan(ret.imag())){
453  ostringstream oss;
454  oss << "expint: nan encountered w n = " << n
455  << " & z = " << z;
456  throw std::runtime_error(oss.str());
457  }
458  return ret;
459  }
