130  complex<RealType> cf_fourier_x (RealType omega
131  ) const
132  {
133  omega *= exp(mu())/_raw_stddev;
134  complex<RealType> ret;
135  if (omega == 0) {
136  ret= static_cast<complex<RealType> >(1);
137  }
138  else {
139  // First deal with the near Dirac delta near the origin
140  RealType lo_x = 1e-9;
141  ret = boost::math::cdf(ln_dist, lo_x);
142  // cap the upper range when pdf is samll or frequency is high enough
143  // to assure cancelation.
144  const RealType eps = std::numeric_limits<RealType>::epsilon();
145  const RealType tol = boost::math::tools::root_epsilon<RealType>();
146  RealType hi_x = boost::math::quantile(boost::math::complement(ln_dist,eps));
147  hi_x = std::min(hi_x, 1/(fabs(omega)*tol));
148  // the integration range will be mapped to y in [0,1]
149  RealType lo_y=sqrt(lo_x)/(1+sqrt(lo_x));
150  RealType hi_y=sqrt(hi_x)/(1+sqrt(hi_x));
151  CFIntegrandX cf_integrand( omega, sigma());
152  const int kronrod_order = 15;
153  unsigned max_depth = 16;
154  ret += (hi_y > lo_y) ? gauss_kronrod<RealType, kronrod_order>::integrate(cf_integrand, lo_y, hi_y,
155  max_depth, tol)
156  :0;
157  }

297  static complex<RealType> LambertW(complex<RealType> z, int k = 0)
298  {
299  //For some particular z and k W(z,k) has simple value:
300  if (z == 0.) return (k == 0) ? 0. : -INFINITY;
301  if (z == -exp(-1.) && (k == 0 || k == -1)) return -1.;
302  if (z == exp(1.) && k == 0) return 1.;
303 
304  //Halley method begins
305  complex<RealType> w{ InitPoint(z, k) }, wprev{ InitPoint(z, k) }; // intermediate values in the Halley method
306  const unsigned int maxiter = 30; // max number of iterations. This eliminates improbable infinite loops
307  unsigned int iter = 0; // iteration counter
308  RealType prec = 1.E-30; // difference threshold between the last two iteration results (or the iter number of iterations is taken)
309 
310  do
311  {
312  wprev = w;
313  w -= 2.*((zexpz(w) - z) * zexpz_d(w)) /
314  (2.*pow(zexpz_d(w),2) - (zexpz(w) - z)*zexpz_dd(w));
315  iter++;
316  } while ((abs(w - wprev) > prec) && iter < maxiter);
317  return w;
318  }
